class Solution:
    def myAtoi(self, s: str) -> int:
        # Step 1: Remove leading whitespace
        s = s.lstrip()

        if not s:
            return 0

        # Step 2: Handle sign
        sign = 1
        i = 0
        if s[0] == '-':
            sign = -1
            i += 1
        elif s[0] == '+':
            i += 1

        # Step 3: Convert digits to integer
        result = 0
        while i < len(s) and s[i].isdigit():
            result = result * 10 + int(s[i])
            i += 1

        # Step 4: Apply sign and handle 32-bit overflow
        result *= sign
        int_min, int_max = -2**31, 2**31 - 1
        if result < int_min:
            return int_min
        elif result > int_max:
            return int_max
        else:
            return result
